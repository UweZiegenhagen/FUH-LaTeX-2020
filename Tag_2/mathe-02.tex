\begin{align} % jede Zeile erhält eine Gleichungsnummer
a &= xyz \cdot y \\
a^2 &= x \cdot y^2 
\end{align}

\begin{align*} % keine Zeile erhält eine Gleichungsnummer
	a &= xyz \cdot y \\
	a^2 &= x \cdot y^2 
\end{align*}


\begin{alignat}{3}
	a       &= xyz \cdot y &&= 3.1415927 \\
	a^2 &= x \cdot y^2 &&= 2.785487487497498498 
\end{alignat}

\begin{alignat}{4}
	a       &= xyz \cdot y &&= 3.1415927 &&&= 123 \\
	a^2 &= x \cdot y^2 &&= 2.785487487497498498 &&&= 432 
\end{alignat}
